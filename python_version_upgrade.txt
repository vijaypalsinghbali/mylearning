

HOW TO UPDATE PYTHON VERSION:


apt install software-properties-common
add-apt-repository ppa:deadsnakes/ppa
apt install python3.12
Done now remove old versions
cd /etc/
ls -ltrh|grep python
ls /usr/bin/python*
ls -ltrh /usr/bin/python*
apt remove python3.8
apt autoremove
rm /usr/bin/python3.8
sudo ln -s /usr/local/bin/python3.12 /usr/bin/python3
ln -s /usr/bin/python3.12 /usr/bin/python

Path to base image for a Dockerfile:




Use this dockerfile to create image (explore-feed)

FROM ubuntu:20.04
RUN apt-get update && apt-get install -y software-properties-common
RUN apt-get update --fix-missing

# update pip
RUN apt-get update
RUN apt-get install -y python3.12-distutils
RUN python3.12 -m ensurepip --upgrade
RUN python3.12 -m pip install --upgrade pip setuptools wheel cython
RUN ln -svf /usr/bin/python3.12 /usr/bin/python
RUN pip3 install wheel requests uwsgi six
RUN pip3 install circus==0.16.1

# RUN pip3 install scipy==1.4.1
